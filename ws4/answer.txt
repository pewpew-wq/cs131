show ()
{
        echo "\$0 = $0, \$1 = $1, \$2 = $2"
        echo "$# servers exist and look like $*"

        local count=1;

        while [[ $# -gt 0 ]]; do
                echo "$count: $1..."

                if ping -c 1 -W 2 "$1" > /dev/null 2>&1; then
                        echo "$1 is online"
                else
                        echo "$1 is offline"
                fi

                count=$((count + 1))
                echo "finished #$count"
                shift
        done
}


checks list of server addresses if they're online, ($0, $1, $2) holds environment arguments ($#) counts total number of arguments ($*) puts all arguments together

the while loop iterates through the arguments until it's 0 then the body if statements sends 1 ping (-c 1) and waits 2 seconds (-W 2) to the target ($1) as (> /dev/null 2>&1) makes the output clean so if it pings it returns online otherwise offline (shift) moves through the arguments
